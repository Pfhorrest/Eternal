function got_item(item, player)	if item == _item_magnum then		if count_item(player, _item_magnum) > 2 then			remove_item(player, _item_magnum)			add_item(player, _item_magnum_magazine)		end	elseif item == _item_plasma_pistol then		if count_item(player, _item_plasma_pistol) > 1 then			remove_item(player, _item_plasma_pistol)			add_item(player, _item_magnum_magazine)		end	elseif item == _item_assault_rifle then		if count_item(player, _item_assault_rifle) > 1 then			remove_item(player, _item_assault_rifle)			add_item(player, _item_assault_rifle_magazine)		end	elseif item == _item_missile_launcher then		if count_item(player, _item_missile_launcher) > 1 then			remove_item(player, _item_missile_launcher)			add_item(player, _item_missile_launcher_magazine)			add_item(player, _item_assault_grenade_magazine)		end	elseif item == _item_alien_weapon then		if count_item(player, _item_alien_weapon) > 1 then			remove_item(player, _item_alien_weapon)			add_item(player, _item_shotgun_magazine)		end	elseif item == _item_flamethrower then		if count_item(player, _item_flamethrower) > 1 then			remove_item(player, _item_flamethrower)			add_item(player, _item_flamethrower_canister)		end	elseif item == _item_shotgun then		if count_item(player, _item_shotgun) > 2 then			remove_item(player, _item_shotgun)			add_item(player, _item_smg_ammo)		end	elseif item == _item_smg then		if count_item(player, _item_smg) > 1 then			remove_item(player, _item_smg)			add_item(player, _item_smg_ammo)		end	elseif item == _item_assault_rifle_magazine then		if count_item(player, _item_assault_rifle) < 1 then			remove_item(player, _item_assault_rifle_magazine)			add_item(player, _item_assault_rifle)		end	elseif item == _item_missile_launcher_magazine then		if count_item(player, _item_missile_launcher) < 1 then			remove_item(player, _item_missile_launcher_magazine)			add_item(player, _item_missile_launcher)		else			add_item(player, _item_assault_grenade_magazine)		end	elseif item == _item_shotgun_magazine then		if count_item(player, _item_alien_weapon) < 1 then			remove_item(player, _item_shotgun_magazine)			add_item(player, _item_alien_weapon)		end	elseif item == _item_flamethrower_canister then		if count_item(player, _item_flamethrower) < 1 then			remove_item(player, _item_flamethrower_canister)			add_item(player, _item_flamethrower)		end	endend
